<!DOCTYPE html>
<html ng-app="myAppWithSceDisabledmyApp">
  <head>
    <meta charset="utf-8">
    <title>$sce disabled demo</title>
    <link rel="stylesheet" href="../../../../../platform/css/bootstrap/css/bootstrap.css" type="text/css"/>
    <style type="text/css">
      body{
        padding: 10px;
      }
    </style>
  </head>
  <body>
    <div ng-controller="SceDemoCtrl">
      <textarea ng-model="userHtml" style="width:500px;"> </textarea>
      <div ng-bind-html="userHtml"></div>
    </div>
    <script type="text/javascript" src="../../angular.js"></script>
    <script type="text/javascript">
          angular.module('myAppWithSceDisabledmyApp', []).config(function($sceProvider) {
            // Completely disable SCE. For demonstration purposes only!
            // Do not use in new projects.
            $sceProvider.enabled(false);
          }).controller('SceDemoCtrl', function($scope) {
            $scope.userHtml = '<div style="color:green;" class="ng-demo" custom-attr="1">把$sce设为disabled后的处理,userHtml可以是任意的html代码，包括style但不包括script（会报错）</div><style type="text/css">.ng-demo{border:1px solid orange;}</style>';
          });
    </script>
  </body>
</html>
