<!DOCTYPE html>
<html ng-app="controllerApp">
  <head>
    <meta charset="utf-8">
    <title>ng controller demo</title>
    <link rel="stylesheet" href="../../../../../platform/css/bootstrap/css/bootstrap.css" type="text/css"/>
    <style type="text/css">
      body{
        padding: 10px;
      }
      div.spicy div {
        padding: 10px;
        border: solid 2px blue;
      }
    </style>
  </head>
  <body>
    <div ng-controller="SpicyCtrl">
      <input ng-model="customSpice">
      <button ng-click="spicy('chili')">Chili</button>
      <button ng-click="spicy(customSpice)">Custom spice</button>
      <p>The food is {{spice}} spicy!</p>
    </div>

    <div class="spicy">
      <div ng-controller="MainCtrl">
        <p>Good {{timeOfDay}}, {{name}}!</p>
        <div ng-controller="ChildCtrl">
          <p>Good {{timeOfDay}}, {{name}}!</p>
          <div ng-controller="GrandChildCtrl">
            <p>Good {{timeOfDay}}, {{name}}!</p>
          </div>
        </div>
      </div>
    </div>
    <script type="text/javascript" src="../../angular.js"></script>
    <script type="text/javascript">
      var myApp = angular.module('controllerApp', []);
      //scope 定义的函数可以传递参数
      myApp.controller('SpicyCtrl', ['$scope', function($scope){
          $scope.customSpice = "wasabi";
          $scope.spice = 'very';
          $scope.spicy = function(spice){
            $scope.spice = spice;
          };
      }]);
    
      //层级显示
      myApp.controller('MainCtrl', ['$scope', function($scope){
          $scope.timeOfDay = 'morning';
          $scope.name = 'Nikki';
      }]);
      myApp.controller('ChildCtrl', ['$scope', function($scope){
          $scope.name = 'Mattie';
      }]);
      myApp.controller('GrandChildCtrl', ['$scope', function($scope){
          $scope.timeOfDay = 'evening';
          $scope.name = 'Gingerbreak Baby';
      }]);
    </script>
  </body>
</html>
